# p1-compute min of 3 nr

a <=> int(input("Input first nr: "))
b <=> int(input("Input second nr: "))
c <=> int(input("Input third nr: "))
minimum <=> a
if b < minimum:
    minimum <=> b
if c < minimum:
    minimum <=> c
display(minimum)

= becomes <=>
== becomes =:=
display() used for printing to the console

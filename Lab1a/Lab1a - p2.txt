# p2- gcd of 2 nr

a <=> int(input("First nr: "))
b <=> int(input("Second nr: "))
if a > b:
    minimum <=> b
else:
    minimum <=> a
for i in range(1, minimum + 1):
    if (a % i =:= 0) and (b % i =:= 0):
        gcd <=> i
display(gcd)
